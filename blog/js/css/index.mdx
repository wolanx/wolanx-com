## 光晕效果


export const GuangYun = () => {
    function doUpdate(e) {
        let r = 0
        let isTransition = true
        if (isTransition) {
            let animateTime = setInterval(function () {
                console.log(123);
                if (r >= 153) {
                    clearInterval(animateTime);
                    isTransition = true;
                } else {
                    isTransition = false;
                    r++;
                    e.target.style.webkitMask = '-webkit-gradient(radial, 45 25, ' + r + ', 45 25, ' + (r + 15) + ', from(rgb(0, 0, 0)), color-stop(0.5, rgba(0, 0, 0, 0.2)), to(rgb(0, 0, 0)))'
                }
            }, 5);
        }
    }
    return (
        <div onMouseEnter={doUpdate} style={{width: 500, height: 500, background: '#ccc'}}>光晕 hover</div>
    )
}
